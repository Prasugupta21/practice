#include<iostream>
#include<cstring>
using namespace std;







void checkPalindrome(char str[]) {
	int flag=0;
	int x=strlen(str);
		int i=0;
	int j=x-1;
	
		while((str[i]==str[j])&&(i<=j))
		{
          flag=1;
		  i++;
		  j--;
                
		}

		if(flag==1)
		{
			cout<<"YES";
		}
		else
		{
			cout<<"NO";
		}
	
	
	
	
	


	
	
	
	
		
}
int main()
{
	char str[100];
	cin>>str;
	checkPalindrome(str);
	
	return 0;
}
